// Limpiar selección de estudiante function limpiarSeleccionEstudiante() { document.getElementById('student-id').value = ''; document.getElementById('foto-estudiante-container').style.display = 'none'; document.querySelector('button[type="submit"]').disabled = true; estudianteSeleccionado = null; } // Configurar editores de texto enriquecido function configurarEditoresTexto() { const editores = ['apoyo-editor', 'acciones-editor', 'recomendaciones-editor', 'nota-confidencial-editor']; editores.forEach(editorId => { const editor = document.getElementById(editorId); const toolbar = document.querySelector(`[data-target="${editorId}"]`); if (!editor || !toolbar) return; configurarToolbar(toolbar, editor); editor.addEventListener('input', function() { actualizarEstadoEditor(this); validarFormularioCompleto(); }); editor.addEventListener('focus', function() { this.parentElement.classList.add('focused'); }); editor.addEventListener('blur', function() { this.parentElement.classList.remove('focused'); actualizarEstadoEditor(this); }); editor.addEventListener('paste', function(e) { e.preventDefault(); const texto = (e.clipboardData || window.clipboardData).getData('text/plain'); document.execCommand('insertText', false, texto); }); actualizarEstadoEditor(editor); }); } // Configurar toolbar de editor function configurarToolbar(toolbar, editor) { const botones = toolbar.querySelectorAll('button[data-command]'); setTimeout(() => { actualizarEstadoEditor(editor); validarFormularioCompleto(); }, 10); }); }); editor.addEventListener('selectionchange', () => actualizarEstadoBotones(toolbar)); editor.addEventListener('keyup', () => actualizarEstadoBotones(toolbar)); editor.addEventListener('mouseup', () => actualizarEstadoBotones(toolbar)); } // Actualizar estado de botones del toolbar function actualizarEstadoBotones(toolbar) { const botones = toolbar.querySelectorAll('button[data-command]'); } }); } // Actualizar estado visual del editor function actualizarEstadoEditor(editor) { const contenido = editor.innerHTML.trim(); const soloEspacios = editor.textContent.trim() === ''; const esNotaConfidencial = editor.id === 'nota-confidencial-editor' || editor.id === 'editar-nota-confidencial-editor'; editor.classList.remove('has-content', 'required-empty'); if (soloEspacios || contenido === '') { if (!esNotaConfidencial) { editor.classList.add('required-empty'); } } else { editor.classList.add('has-content'); } } // Validar formulario completo function validarFormularioCompleto() { const studentId = document.getElementById('student-id').value; const apoyoValido = validarEditor('apoyo-editor'); const accionesValido = validarEditor('acciones-editor'); const recomendacionesValido = validarEditor('recomendaciones-editor'); const formularioValido = studentId && apoyoValido && accionesValido && recomendacionesValido; const botonGuardar = document.querySelector('button[type="submit"]'); botonGuardar.disabled = !formularioValido; return formularioValido; } // Validar editor específico function validarEditor(editorId) { const editor = document.getElementById(editorId); if (!editor) return false; const contenido = editor.textContent.trim(); return contenido.length > 0; } // Obtener contenido de editor function obtenerContenidoEditor(editorId) { const editor = document.getElementById(editorId); if (!editor) return ''; return editor.innerHTML.trim(); } // Limpiar formulario function limpiarFormulario() { if (confirm('¿Estás seguro de que deseas limpiar todos los campos?')) { document.getElementById('form-eae').reset(); limpiarSeleccionEstudiante(); document.getElementById('estudiante-search').value = ''; document.getElementById('estudiantes-dropdown').style.display = 'none'; limpiarEditoresTexto(); validarFormularioCompleto(); } } // Limpiar editores de texto function limpiarEditoresTexto() { const editores = ['apoyo-editor', 'acciones-editor', 'recomendaciones-editor', 'nota-confidencial-editor']; editores.forEach(editorId => { const editor = document.getElementById(editorId); if (editor) { editor.innerHTML = ''; actualizarEstadoEditor(editor); } }); } // Validar y guardar formulario async function validarYGuardarFormulario() { if (enviandoFormulario) { console.log('Envío bloqueado: formulario ya en proceso'); showMessage('Por favor espere, el registro se está guardando...', 'warning'); return; } const botonGuardar = document.querySelector('button[type="submit"]'); if (botonGuardar.disabled) { console.log('Botón ya deshabilitado, ignorando clic'); return; } if (!validarFormularioCompleto()) { showMessage('Por favor, completa todos los campos obligatorios', 'warning'); return; } enviandoFormulario = true; const datosFormulario = { student_id: parseInt(document.getElementById('student-id').value), supports: obtenerContenidoEditor('apoyo-editor'), actions: obtenerContenidoEditor('acciones-editor'), recommendations: obtenerContenidoEditor('recomendaciones-editor'), confidential_note: obtenerContenidoEditor('nota-confidencial-editor') }; if (!datosFormulario.supports || !datosFormulario.actions || !datosFormulario.recommendations) { showMessage('Los campos Apoyo, Acciones y Recomendaciones son obligatorios', 'warning'); enviandoFormulario = false; return; } try { // Obtener worker_id del usuario actual const session = getStoredSession(); const workerData = await supabaseRequest(`/workers?select=worker_id&email=eq.${session.user.user_mail}&limit=1`); if (workerData.length === 0) { throw new Error('No se pudo identificar al usuario actual'); } const workerId = workerData[0].worker_id; // Preparar datos para inserción const insertData = { student_id: datosFormulario.student_id, worker_id: workerId, topic_date: new Date().toISOString().split('T')[0], supports: datosFormulario.supports, actions: datosFormulario.actions, recommendations: datosFormulario.recommendations, confidential_note: datosFormulario.confidential_note || null, is_open: 'SI' }; botonGuardar.disabled = true; botonGuardar.innerHTML = 'Guardando...'; // Insertar registro await supabaseRequest('/stm_eae_topics', { method: 'POST', body: JSON.stringify(insertData) }); // Enviar notificación al director de sección await enviarNotificacionDirector(datosFormulario.student_id, datosFormulario); showMessage('Registro EAE guardado exitosamente', 'success'); setTimeout(() => { limpiarFormularioCompleto(); showMessage('Formulario listo para un nuevo registro', 'info'); }, 2000); } catch (error) { console.error('Error al guardar:', error); showMessage('Error al guardar: ' + error.message, 'danger'); restaurarBotonGuardar(); } finally { enviandoFormulario = false; } } // Enviar notificación al director de sección async function enviarNotificacionDirector(studentId, datosFormulario) { try { // Obtener datos del estudiante y director const studentData = await supabaseRequest(`/students?select=student_first_name,student_last_name_1,courses!inner(course_name,grades!inner(sections!inner(director_email)))&student_id=eq.${studentId}&limit=1`); if (studentData.length > 0 && studentData[0].courses?.grades?.sections?.director_email) { const estudiante = studentData[0]; const directorEmail = estudiante.courses.grades.sections.director_email; const asunto = 'El E.A.E. ha registrado un asunto con un estudiante de tu sección'; const cuerpo = `

Nuevo Registro EAE

En relación con el estudiante ${estudiante.student_first_name} ${estudiante.student_last_name_1} del curso ${estudiante.courses.course_name} se ha registrado el siguiente asunto:

Apoyo:
${datosFormulario.supports}

Acciones:
${datosFormulario.actions}

Recomendaciones:
${datosFormulario.recommendations}

Recuerda tenerlo en cuenta en los seguimientos.

Este es un correo enviado de forma automática por SchoolNet. Por favor no responder.

`; await sendNotification(directorEmail, asunto, cuerpo, true); } } catch (error) { console.error('Error enviando notificación (no crítico):', error); } } // Restaurar botón guardar function restaurarBotonGuardar() { const botonGuardar = document.querySelector('button[type="submit"]'); botonGuardar.disabled = false; botonGuardar.innerHTML = 'Guardar'; } // Limpiar formulario completo function limpiarFormularioCompleto() { document.getElementById('form-eae').reset(); limpiarSeleccionEstudiante(); document.getElementById('estudiante-search').value = ''; document.getElementById('estudiantes-dropdown').style.display = 'none'; limpiarEditoresTexto(); validarFormularioCompleto(); } // Variables para el modal de registros let paginaActualRegistros = 1; let datosRegistrosActuales = null; // Abrir modal de registros anteriores function abrirModalRegistros() { const modal = document.getElementById('modal-registros'); modal.style.display = 'flex'; paginaActualRegistros = 1; cargarRegistrosEAE(1); } // Cerrar modal de registros function cerrarModalRegistros() { const modal = document.getElementById('modal-registros'); modal.style.display = 'none'; datosRegistrosActuales = null; paginaActualRegistros = 1; } // Cargar registros EAE del usuario actual async function cargarRegistrosEAE(pagina = 1) { mostrarLoadingModal(); try { // Obtener worker_id del usuario actual const session = getStoredSession(); const workerData = await supabaseRequest(`/workers?select=worker_id&email=eq.${session.user.user_mail}&limit=1`); if (workerData.length === 0) { throw new Error('No se pudo identificar al usuario actual'); } const workerId = workerData[0].worker_id; const registrosPorPagina = 25; const offset = (pagina - 1) * registrosPorPagina; // Consulta para obtener registros con paginación const registros = await supabaseRequest(`/stm_eae_topics?select=eae_topic_id,topic_date,supports,actions,recommendations,confidential_note,students!inner(student_first_name,student_last_name_1,student_last_name_2,courses(course_name))&worker_id=eq.${workerId}&is_open=eq.SI&order=topic_date.desc,eae_topic_id.desc&limit=${registrosPorPagina}&offset=${offset}`); // Contar total de registros const countResult = await supabaseRequest(`/stm_eae_topics?select=eae_topic_id&worker_id=eq.${workerId}&is_open=eq.SI`); const totalRegistros = countResult.length; // Procesar registros para crear resúmenes const registrosProcesados = registros.map(registro => { const estudiante = registro.students; const nombreCompleto = `${estudiante.student_first_name} ${estudiante.student_last_name_1} ${estudiante.student_last_name_2 || ''}`.trim(); return { eae_topic_id: registro.eae_topic_id, student_name: nombreCompleto, course_name: estudiante.courses?.course_name || '', topic_date_formatted: formatDate(registro.topic_date), supports_summary: truncateText(registro.supports, 50), actions_summary: truncateText(registro.actions, 50), recommendations_summary: truncateText(registro.recommendations, 50), confidential_note_summary: truncateText(registro.confidential_note, 50) }; }); const totalPaginas = Math.ceil(totalRegistros / registrosPorPagina); datosRegistrosActuales = { registros: registrosProcesados, paginacion: { paginaActual: pagina, totalPaginas: totalPaginas, totalRegistros: totalRegistros, registrosPorPagina: registrosPorPagina } }; ocultarLoadingModal(); if (registrosProcesados.length === 0) { mostrarSinRegistros(); } else { mostrarRegistros(datosRegistrosActuales); } } catch (error) { ocultarLoadingModal(); console.error('Error al cargar registros:', error); mostrarErrorModal('Error al cargar registros: ' + error.message); } } // Función auxiliar para truncar texto function truncateText(text, maxWords) { if (!text || text.trim() === '') return ''; const plainText = text.replace(/<[^>]*>/g, ' ').replace(/\s+/g, ' ').trim(); const words = plainText.split(' '); if (words.length <= maxWords) return plainText; return words.slice(0, maxWords).join(' ') + '...'; } // Mostrar loading del modal function mostrarLoadingModal() { document.getElementById('modal-loading').style.display = 'block'; document.getElementById('registros-content').style.display = 'none'; document.getElementById('sin-registros').style.display = 'none'; } // Ocultar loading del modal function ocultarLoadingModal() { document.getElementById('modal-loading').style.display = 'none'; } // Mostrar mensaje de sin registros function mostrarSinRegistros() { document.getElementById('registros-content').style.display = 'none'; document.getElementById('sin-registros').style.display = 'block'; } // Mostrar error en el modal function mostrarErrorModal(mensaje) { const sinRegistros = document.getElementById('sin-registros'); sinRegistros.innerHTML = `

Error

${mensaje}

`; sinRegistros.style.display = 'block'; document.getElementById('registros-content').style.display = 'none'; } // Mostrar los registros en la tabla function mostrarRegistros(datos) { document.getElementById('registros-content').style.display = 'block'; const { registros, paginacion } = datos; const inicio = (paginacion.paginaActual - 1) * paginacion.registrosPorPagina + 1; const fin = Math.min(inicio + registros.length - 1, paginacion.totalRegistros); // Actualizar información de registros const infoRegistros = document.getElementById('info-registros'); infoRegistros.textContent = `Mostrando ${inicio}-${fin} de ${paginacion.totalRegistros} registros`; // Renderizar tabla renderizarTablaRegistros(registros); // Actualizar paginación actualizarPaginacionRegistros(paginacion); } // Renderizar la tabla de registros function renderizarTablaRegistros(registros) { const tbody = document.getElementById('tabla-registros-body'); if (registros.length === 0) { tbody.innerHTML = 'No se encontraron registros'; return; } const fragment = document.createDocumentFragment(); registros.forEach(registro => { const tr = document.createElement('tr'); tr.innerHTML = ` ${escapeHtml(registro.student_name)} ${registro.course_name ? `
${escapeHtml(registro.course_name)}` : ''} ${registro.topic_date_formatted} ${escapeHtml(registro.supports_summary)} ${escapeHtml(registro.actions_summary)} ${escapeHtml(registro.recommendations_summary)} ${registro.confidential_note_summary ? escapeHtml(registro.confidential_note_summary) : 'Sin nota'} `; fragment.appendChild(tr); }); tbody.innerHTML = ''; tbody.appendChild(fragment); } // Actualizar controles de paginación function actualizarPaginacionRegistros(paginacion) { const { paginaActual, totalPaginas } = paginacion; let paginacionHTML = ''; if (totalPaginas > 1) { // Botón anterior if (paginaActual > 1) { paginacionHTML += ``; } // Números de página const startPage = Math.max(1, paginaActual - 2); const endPage = Math.min(totalPaginas, paginaActual + 2); if (startPage > 1) { paginacionHTML += ``; if (startPage > 2) { paginacionHTML += `...`; } } for (let i = startPage; i <= endPage; i++) { const isActive = i === paginaActual; const btnClass = isActive ? 'btn-primary' : 'btn-secondary'; paginacionHTML += ``; } if (endPage < totalPaginas) { if (endPage < totalPaginas - 1) { paginacionHTML += `...`; } paginacionHTML += ``; } // Botón siguiente if (paginaActual < totalPaginas) { paginacionHTML += ``; } } // Actualizar ambos contenedores de paginación document.getElementById('paginacion-top').innerHTML = paginacionHTML; document.getElementById('paginacion-bottom').innerHTML = paginacionHTML; } // Cambiar a una página específica de registros function cambiarPaginaRegistros(nuevaPagina) { if (nuevaPagina !== paginaActualRegistros) { cargarRegistrosEAE(nuevaPagina); } } // Función auxiliar para escapar HTML function escapeHtml(text) { if (!text) return ''; const div = document.createElement('div'); div.textContent = text; return div.innerHTML; } // Variables para el modal de edición let registroEnEdicion = null; // Abrir modal para editar un registro async function editarRegistro(eaeTopicId) { try { // Obtener datos del registro específico const registroData = await supabaseRequest(`/stm_eae_topics?select=eae_topic_id,topic_date,supports,actions,recommendations,confidential_note,students!inner(student_first_name,student_last_name_1,student_last_name_2,courses(course_name)),worker_id&eae_topic_id=eq.${eaeTopicId}&is_open=eq.SI&limit=1`); if (registroData.length === 0) { showMessage('Registro no encontrado o no disponible', 'danger'); return; } const registro = registroData[0]; // Verificar que el registro pertenece al usuario actual const session = getStoredSession(); const workerData = await supabaseRequest(`/workers?select=worker_id&email=eq.${session.user.user_mail}&limit=1`); if (workerData.length === 0 || workerData[0].worker_id !== registro.worker_id) { showMessage('No tienes permisos para editar este registro', 'danger'); return; } // Procesar datos del estudiante const estudiante = registro.students; const nombreCompleto = `${estudiante.student_first_name} ${estudiante.student_last_name_1} ${estudiante.student_last_name_2 || ''}`.trim(); mostrarModalEdicion(registroEnEdicion); } catch (error) { console.error('Error al cargar registro para editar:', error); showMessage('Error al cargar registro: ' + error.message, 'danger'); } } // Mostrar el modal de edición con los datos del registro function mostrarModalEdicion(registro) { // Configurar información del estudiante const infoEstudiante = document.getElementById('info-estudiante-editar'); ${escapeHtml(registro.student_name)}

${registro.course_name ? escapeHtml(registro.course_name) + ' • ' : ''}Fecha: ${registro.topic_date_formatted}

`; // Establecer ID del registro document.getElementById('eae-topic-id').value = registro.eae_topic_id; // Llenar editores con contenido actual document.getElementById('editar-apoyo-editor').innerHTML = registro.supports || ''; document.getElementById('editar-acciones-editor').innerHTML = registro.actions || ''; document.getElementById('editar-recomendaciones-editor').innerHTML = registro.recommendations || ''; document.getElementById('editar-nota-confidencial-editor').innerHTML = registro.confidential_note || ''; // Configurar editores de texto configurarEditoresEdicion(); // Mostrar modal document.getElementById('modal-editar-registro').style.display = 'flex'; // Validar formulario inicial validarFormularioEdicion(); } // Configurar los editores de texto del modal de edición function configurarEditoresEdicion() { const editores = ['editar-apoyo-editor', 'editar-acciones-editor', 'editar-recomendaciones-editor', 'editar-nota-confidencial-editor']; editores.forEach(editorId => { const editor = document.getElementById(editorId); const toolbar = document.querySelector(`[data-target="${editorId}"]`); if (!editor || !toolbar) return; // Remover event listeners previos para evitar duplicados const nuevoEditor = editor.cloneNode(true); editor.parentNode.replaceChild(nuevoEditor, editor); // Configurar eventos del toolbar configurarToolbar(toolbar, nuevoEditor); // Eventos del editor nuevoEditor.addEventListener('input', function() { actualizarEstadoEditor(this); validarFormularioEdicion(); }); nuevoEditor.addEventListener('focus', function() { this.parentElement.classList.add('focused'); }); nuevoEditor.addEventListener('blur', function() { this.parentElement.classList.remove('focused'); actualizarEstadoEditor(this); }); // Manejar paste nuevoEditor.addEventListener('paste', function(e) { e.preventDefault(); const texto = (e.clipboardData || window.clipboardData).getData('text/plain'); document.execCommand('insertText', false, texto); }); // Inicializar estado actualizarEstadoEditor(nuevoEditor); }); } // Validar el formulario de edición function validarFormularioEdicion() { const apoyoValido = validarEditorEdicion('editar-apoyo-editor'); const botonGuardar = document.getElementById('btn-guardar-edicion'); botonGuardar.disabled = !formularioValido; return formularioValido; } // Validar un editor específico de edición function validarEditorEdicion(editorId) { const editor = document.getElementById(editorId); if (!editor) return false; const contenido = editor.textContent.trim(); return contenido.length > 0; } // Obtener contenido de editor de edición function obtenerContenidoEditorEdicion(editorId) { const editor = document.getElementById(editorId); if (!editor) return ''; return editor.innerHTML.trim(); } // Guardar los cambios del registro editado async function guardarEdicionRegistro() { const botonGuardar = document.getElementById('btn-guardar-edicion'); const datosActualizados = { supports: obtenerContenidoEditorEdicion('editar-apoyo-editor'), actions: obtenerContenidoEditorEdicion('editar-acciones-editor'), recommendations: obtenerContenidoEditorEdicion('editar-recomendaciones-editor'), confidential_note: obtenerContenidoEditorEdicion('editar-nota-confidencial-editor') || null }; if (!datosActualizados.supports || !datosActualizados.actions || !datosActualizados.recommendations) { showMessage('Los campos Apoyo, Acciones y Recomendaciones son obligatorios', 'warning'); return; } try { botonGuardar.disabled = true; botonGuardar.innerHTML = 'Guardando...'; // Actualizar el registro await supabaseRequest(`/stm_eae_topics?eae_topic_id=eq.${eaeTopicId}`, { method: 'PATCH', body: JSON.stringify(datosActualizados) }); cerrarModalEditar(); showMessage('Registro actualizado exitosamente', 'success'); // Recargar registros si el modal está abierto if (document.getElementById('modal-registros').style.display === 'flex') { cargarRegistrosEAE(paginaActualRegistros); } } catch (error) { console.error('Error al guardar:', error); showMessage('Error al guardar: ' + error.message, 'danger'); botonGuardar.disabled = false; botonGuardar.innerHTML = 'Guardar Cambios'; } } // Cerrar el modal de edición function cerrarModalEditar() { document.getElementById('modal-editar-registro').style.display = 'none'; registroEnEdicion = null; // Restaurar botón const botonGuardar = document.getElementById('btn-guardar-edicion'); botonGuardar.disabled = false; botonGuardar.innerHTML = 'Guardar Cambios'; // Limpiar editores const editores = ['editar-apoyo-editor', 'editar-acciones-editor', 'editar-recomendaciones-editor', 'editar-nota-confidencial-editor']; editores.forEach(editorId => { const editor = document.getElementById(editorId); if (editor) { editor.innerHTML = ''; } }); }
