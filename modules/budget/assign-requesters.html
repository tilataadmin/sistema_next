// ===== EVENT HANDLERS ===== /** * Maneja el cambio de año académico */ async function handleYearChange() { const yearId = uiElements.selectAcademicYear.value; if (!yearId) { resetApplicationState(); populateWorkersSelect(); return; } try { // Actualizar estado global globalData.selectedYear = yearId; globalData.selectedWorker = null; // Actualizar información del año const yearInfo = getAcademicYearInfo(yearId); // Cargar asignaciones del usuario para este año await loadUserAssignments(yearId); logDebug('Año académico cambiado', { yearId, yearName: yearInfo?.year_name }); } catch (error) { console.error('Error al cambiar año académico:', error); showMessage('Error al cargar datos del año académico: ' + error.message, 'danger'); } } /** * Maneja el cambio de trabajador */ async function handleWorkerChange() { const workerEmail = uiElements.selectWorker.value; if (!workerEmail || !globalData.selectedYear) { resetAssignmentsView(); return; } try { // Actualizar estado global globalData.selectedWorker = workerEmail; // Mostrar loading en la tabla showAssignmentsLoading(); // Cargar autorizaciones actuales del trabajador await loadCurrentAuthorizations(workerEmail, globalData.selectedYear); // Renderizar tabla de asignaciones renderAssignmentsTable(); // Mostrar secciones de control uiElements.summaryInfo.classList.remove('hidden'); uiElements.selectAllSection.classList.remove('hidden'); uiElements.actionButtons.classList.remove('hidden'); // Actualizar resumen estadístico updateSummaryStatistics(); const workerInfo = getWorkerInfo(workerEmail); logDebug('Trabajador cambiado', { workerEmail, workerName: workerInfo?.full_name, assignmentsCount: globalData.userAssignments.length }); } catch (error) { console.error('Error al cambiar trabajador:', error); showMessage('Error al cargar autorizaciones del trabajador: ' + error.message, 'danger'); resetAssignmentsView(); } } /** * Maneja el checkbox "Seleccionar todo" */ function handleSelectAll() { const isChecked = uiElements.selectAllCheckbox.checked; const checkboxes = document.querySelectorAll('input[data-assignment-id]'); checkboxes.forEach(checkbox => { if (checkbox.checked !== isChecked) { checkbox.checked = isChecked; handleAssignmentCheckboxChange(checkbox); } }); logDebug('Select all toggled', { isChecked, affectedCheckboxes: checkboxes.length }); } /** * Maneja el cambio de un checkbox individual de asignación */ function handleAssignmentCheckboxChange(checkbox) { const assignmentId = checkbox.dataset.assignmentId; const isAuthorized = checkbox.checked; const originallyAuthorized = originalState.has(assignmentId) ? originalState.get(assignmentId) : false; // Actualizar cambios pendientes if (isAuthorized !== originallyAuthorized) { pendingChanges.set(assignmentId, isAuthorized); } else { pendingChanges.delete(assignmentId); } // Actualizar estado visual updateAssignmentRowStatus(assignmentId, isAuthorized); // Actualizar controles y estadísticas updateSelectAllState(); updateSummaryStatistics(); updateSaveButtonState(); logDebug('Assignment checkbox changed', { assignmentId, isAuthorized, originallyAuthorized, pendingChangesCount: pendingChanges.size }); } // ===== CARGA DE DATOS ===== /** * Carga las asignaciones presupuestales del usuario actual */ async function loadUserAssignments(yearId) { try { const session = getStoredSession(); if (!session || !session.user || !session.user.user_id) { throw new Error('No se pudo obtener la información del usuario actual'); } // Obtener email del usuario desde la sesión const userEmail = session.user.user_mail || session.user.email; if (!userEmail) { throw new Error('No se pudo obtener el email del usuario'); } // Consulta optimizada de asignaciones const query = `/budget_assignments?select=assignment_id,budget_item_id,assignment_category,category_detail,approved_value,budget_items(item_name,budget_categories(category_name))&worker_email=eq.${userEmail}&academic_year_id=eq.${yearId}&assignment_status=eq.active&order=budget_items(budget_categories(category_name)).asc,budget_items(item_name).asc`; const assignments = await supabaseRequest(query); // Procesar y enriquecer datos globalData.userAssignments = assignments.map(assignment => ({ assignment_id: assignment.assignment_id, budget_item_id: assignment.budget_item_id, assignment_category: assignment.assignment_category, category_detail: assignment.category_detail, approved_value: assignment.approved_value || 0, item_name: assignment.budget_items?.item_name || 'Sin nombre', category_name: assignment.budget_items?.budget_categories?.category_name || 'Sin categoría', // Campo para tracking de autorización is_authorized: false // Se actualizará al cargar autorizaciones })); console.log(`✅ Asignaciones del usuario cargadas: ${globalData.userAssignments.length} registros`); if (globalData.userAssignments.length === 0) { showMessage('No tienes asignaciones presupuestales para el año seleccionado', 'warning'); } } catch (error) { console.error('Error cargando asignaciones del usuario:', error); globalData.userAssignments = []; throw error; } } /** * Carga las autorizaciones actuales del trabajador para el año específico */ async function loadCurrentAuthorizations(workerEmail, yearId) { try { // Consulta las autorizaciones existentes filtrando por año académico const query = `/budget_requesters?select=assignment_id,budget_assignments!inner(academic_year_id)&worker_email=eq.${workerEmail}&budget_assignments.academic_year_id=eq.${yearId}`; const authorizations = await supabaseRequest(query); // Extraer solo los IDs de asignación autorizadas const authorizedAssignmentIds = authorizations.map(auth => auth.assignment_id); // Actualizar estado de autorización en asignaciones del usuario globalData.userAssignments.forEach(assignment => { assignment.is_authorized = authorizedAssignmentIds.includes(assignment.assignment_id); }); // Guardar estado original para comparación originalState.clear(); globalData.userAssignments.forEach(assignment => { originalState.set(assignment.assignment_id, assignment.is_authorized); }); // Limpiar cambios pendientes pendingChanges.clear(); console.log(`✅ Autorizaciones actuales cargadas: ${authorizedAssignmentIds.length} de ${globalData.userAssignments.length} asignaciones`); } catch (error) { console.error('Error cargando autorizaciones actuales:', error); // En caso de error, marcar todas como no autorizadas globalData.userAssignments.forEach(assignment => { assignment.is_authorized = false; }); originalState.clear(); pendingChanges.clear(); throw error; } } // ===== RENDERIZADO DE INTERFAZ ===== /** * Muestra el estado de loading en la tabla de asignaciones */ function showAssignmentsLoading() { uiElements.assignmentsContainer.innerHTML = `
Cargando...
Cargando asignaciones

Obteniendo autorizaciones del trabajador...

`; } /** * Resetea la vista de asignaciones al estado inicial */ function resetAssignmentsView() { // Ocultar secciones de control uiElements.summaryInfo.classList.add('hidden'); uiElements.selectAllSection.classList.add('hidden'); uiElements.actionButtons.classList.add('hidden'); // Mostrar estado inicial uiElements.assignmentsContainer.innerHTML = `
Selecciona un trabajador

Selecciona el trabajador que deseas autorizar para ver tus asignaciones presupuestales disponibles.

`; // Limpiar estados pendingChanges.clear(); originalState.clear(); globalData.selectedWorker = null; // Deshabilitar botón guardar uiElements.btnSaveChanges.disabled = true; } /** * Renderiza la tabla completa de asignaciones */ function renderAssignmentsTable() { if (globalData.userAssignments.length === 0) { uiElements.assignmentsContainer.innerHTML = `
No hay asignaciones presupuestales

No tienes asignaciones presupuestales activas para el año académico seleccionado.

`; return; } // Crear tabla HTML let tableHTML = ` `; // Agregar filas de asignaciones globalData.userAssignments.forEach(assignment => { const categoryDisplay = assignment.category_detail && assignment.category_detail !== assignment.assignment_category ? `${assignment.assignment_category} - ${assignment.category_detail}` : assignment.assignment_category; const statusClass = assignment.is_authorized ? 'status-authorized' : 'status-not-authorized'; const statusText = assignment.is_authorized ? 'Autorizado' : 'No autorizado'; const statusIcon = assignment.is_authorized ? 'bi-check-circle-fill' : 'bi-circle'; tableHTML += ` `; }); tableHTML += `
Sub-ítem Presupuestal Rubro Categoría Valor Aprobado Estado
${assignment.item_name}
${assignment.category_name}
${categoryDisplay}
 ${formatCurrency(assignment.approved_value)} ${statusText}
`; // Insertar tabla en el contenedor uiElements.assignmentsContainer.innerHTML = tableHTML; // Actualizar estado del checkbox "Seleccionar todo" updateSelectAllState(); } /** * Actualiza el estado visual de una fila de asignación */ function updateAssignmentRowStatus(assignmentId, isAuthorized) { const row = document.querySelector(`tr[data-assignment-id="${assignmentId}"]`); if (!row) return; const statusCell = row.querySelector('.status-column span'); const statusIcon = statusCell.querySelector('i'); if (isAuthorized) { statusCell.className = 'status-authorized'; statusIcon.className = 'bi bi-check-circle-fill me-1'; statusCell.innerHTML = 'Autorizado'; } else { statusCell.className = 'status-not-authorized'; statusIcon.className = 'bi bi-circle me-1'; statusCell.innerHTML = 'No autorizado'; } } /** * Actualiza el estado del checkbox "Seleccionar todo" */ function updateSelectAllState() { const allCheckboxes = document.querySelectorAll('input[data-assignment-id]'); const checkedCheckboxes = document.querySelectorAll('input[data-assignment-id]:checked'); if (allCheckboxes.length === 0) { uiElements.selectAllCheckbox.indeterminate = false; uiElements.selectAllCheckbox.checked = false; return; } if (checkedCheckboxes.length === 0) { uiElements.selectAllCheckbox.indeterminate = false; uiElements.selectAllCheckbox.checked = false; } else if (checkedCheckboxes.length === allCheckboxes.length) { uiElements.selectAllCheckbox.indeterminate = false; uiElements.selectAllCheckbox.checked = true; } else { uiElements.selectAllCheckbox.indeterminate = true; } } // ===== ESTADÍSTICAS Y RESUMEN ===== /** * Actualiza las estadísticas del resumen */ function updateSummaryStatistics() { const totalAssignments = globalData.userAssignments.length; const currentlyAuthorized = document.querySelectorAll('input[data-assignment-id]:checked').length; const pendingAssignments = totalAssignments - currentlyAuthorized; const totalBudget = globalData.userAssignments.reduce((sum, assignment) => sum + (assignment.approved_value || 0), 0); // Actualizar elementos de la interfaz uiElements.totalAssignments.textContent = totalAssignments; uiElements.authorizedAssignments.textContent = currentlyAuthorized; uiElements.pendingAssignments.textContent = pendingAssignments; uiElements.totalBudget.textContent = formatCurrency(totalBudget); logDebug('Summary updated', { total: totalAssignments, authorized: currentlyAuthorized, pending: pendingAssignments, budget: totalBudget }); } /** * Actualiza el estado del botón de guardar */ function updateSaveButtonState() { const hasChanges = pendingChanges.size > 0; uiElements.btnSaveChanges.disabled = !hasChanges; // Actualizar texto del botón if (hasChanges) { const changesCount = pendingChanges.size; uiElements.btnSaveChanges.innerHTML = ` Guardar Autorizaciones (${changesCount} cambios) `; } else { uiElements.btnSaveChanges.innerHTML = ` Guardar Autorizaciones `; } } // ===== HANDLERS DE BOTONES ===== /** * Maneja el evento de guardar cambios */ async function handleSaveChanges() { if (pendingChanges.size === 0) { showMessage('No hay cambios para guardar', 'warning'); return; } if (!globalData.selectedWorker || !globalData.selectedYear) { showMessage('Error: No hay trabajador o año seleccionado', 'danger'); return; } // Mostrar modal de confirmación showConfirmationModal(); } /** * Maneja el evento de cancelar */ function handleCancel() { if (pendingChanges.size > 0) { if (confirm('¿Estás seguro de que quieres descartar los cambios no guardados?')) { revertChanges(); } } else { resetApplicationState(); uiElements.selectAcademicYear.value = ''; uiElements.selectWorker.value = ''; populateWorkersSelect(); } } /** * Maneja el evento de restablecer selección */ function handleResetSelection() { if (pendingChanges.size > 0) { if (confirm('¿Estás seguro de que quieres restablecer todas las selecciones al estado original?')) { revertChanges(); } } } /** * Revierte todos los cambios al estado original */ function revertChanges() { // Restaurar checkboxes al estado original globalData.userAssignments.forEach(assignment => { const checkbox = document.querySelector(`input[data-assignment-id="${assignment.assignment_id}"]`); if (checkbox) { const originalState = originalState.get(assignment.assignment_id) || false; checkbox.checked = originalState; updateAssignmentRowStatus(assignment.assignment_id, originalState); } }); // Limpiar cambios pendientes pendingChanges.clear(); // Actualizar controles updateSelectAllState(); updateSummaryStatistics(); updateSaveButtonState(); showMessage('Cambios revertidos al estado original', 'info'); } // ===== MODAL DE CONFIRMACIÓN ===== /** * Muestra el modal de confirmación con detalles de los cambios */ function showConfirmationModal() { const workerInfo = getWorkerInfo(globalData.selectedWorker); const yearInfo = getAcademicYearInfo(globalData.selectedYear); // Analizar cambios const newAuthorizations = []; const removedAuthorizations = []; pendingChanges.forEach((newState, assignmentId) => { const assignment = globalData.userAssignments.find(a => a.assignment_id === assignmentId); if (assignment) { if (newState) { newAuthorizations.push(assignment); } else { removedAuthorizations.push(assignment); } } }); // Construir mensaje de confirmación let confirmationHTML = `
Trabajador: ${workerInfo ? workerInfo.full_name : 'Desconocido'}
Año Académico: ${yearInfo ? yearInfo.year_name : 'Desconocido'}
Resumen de cambios:
    `; if (newAuthorizations.length > 0) { confirmationHTML += `
  • ${newAuthorizations.length} nueva(s) autorización(es)
    • `; } if (removedAuthorizations.length > 0) { confirmationHTML += `
  • ${removedAuthorizations.length} autorización(es) removida(s)
    • `; } confirmationHTML += `

Estos cambios afectarán las solicitudes de ejecución presupuestal que puede realizar el trabajador.

`; uiElements.confirmMessage.innerHTML = confirmationHTML; // Mostrar modal usando Bootstrap const modal = new bootstrap.Modal(uiElements.confirmModal); modal.show(); } /** * Confirma y ejecuta el guardado de cambios */ async function confirmSaveChanges() { // Cerrar modal const modal = bootstrap.Modal.getInstance(uiElements.confirmModal); if (modal) modal.hide(); try { // Mostrar loading en botón const originalButtonHTML = uiElements.btnSaveChanges.innerHTML; uiElements.btnSaveChanges.disabled = true; uiElements.btnSaveChanges.innerHTML = `
Guardando...
Guardando cambios... `; // Ejecutar guardado await saveAuthorizationChanges(); // Restaurar botón uiElements.btnSaveChanges.innerHTML = originalButtonHTML; uiElements.btnSaveChanges.disabled = false; showMessage('Autorizaciones guardadas correctamente', 'success'); // Actualizar estado original updateOriginalState(); // Limpiar cambios pendientes pendingChanges.clear(); updateSaveButtonState(); } catch (error) { console.error('Error guardando cambios:', error); // Restaurar botón uiElements.btnSaveChanges.innerHTML = originalButtonHTML; uiElements.btnSaveChanges.disabled = false; showMessage('Error al guardar las autorizaciones: ' + error.message, 'danger'); } } // ===== GUARDADO EN BASE DE DATOS ===== /** * Guarda los cambios de autorización en la base de datos */ async function saveAuthorizationChanges() { if (pendingChanges.size === 0) { throw new Error('No hay cambios para guardar'); } // Obtener información del trabajador const workerInfo = getWorkerInfo(globalData.selectedWorker); if (!workerInfo) { throw new Error('No se encontró información del trabajador seleccionado'); } // Preparar listas de cambios const toAuthorize = []; const toRevoke = []; pendingChanges.forEach((isAuthorized, assignmentId) => { if (isAuthorized) { toAuthorize.push(assignmentId); } else { toRevoke.push(assignmentId); } }); // Ejecutar cambios en transacciones separadas para mejor manejo de errores let authorizationsAdded = 0; let authorizationsRemoved = 0; // 1. Remover autorizaciones if (toRevoke.length > 0) { const revokeQuery = `/budget_requesters?assignment_id=in.(${toRevoke.join(',')})&worker_email=eq.${globalData.selectedWorker}`; await supabaseRequest(revokeQuery, { method: 'DELETE' }); authorizationsRemoved = toRevoke.length; } // 2. Agregar nuevas autorizaciones if (toAuthorize.length > 0) { const newAuthorizations = toAuthorize.map(assignmentId => ({ assignment_id: assignmentId, worker_id: workerInfo.worker_id, worker_email: globalData.selectedWorker, authorized_at: new Date().toISOString(), authorized_by: getStoredSession()?.user?.user_id || null })); await supabaseRequest('/budget_requesters', { method: 'POST', body: JSON.stringify(newAuthorizations) }); authorizationsAdded = toAuthorize.length; } logDebug('Authorization changes saved', { added: authorizationsAdded, removed: authorizationsRemoved, workerEmail: globalData.selectedWorker, yearId: globalData.selectedYear }); return { added: authorizationsAdded, removed: authorizationsRemoved }; } /** * Actualiza el estado original después de guardar cambios */ function updateOriginalState() { // Actualizar estado original basado en checkboxes actuales const checkboxes = document.querySelectorAll('input[data-assignment-id]'); checkboxes.forEach(checkbox => { const assignmentId = checkbox.dataset.assignmentId; originalState.set(assignmentId, checkbox.checked); // Actualizar también el estado en globalData const assignment = globalData.userAssignments.find(a => a.assignment_id === assignmentId); if (assignment) { assignment.is_authorized = checkbox.checked; } }); } // ===== VALIDACIONES ===== /** * Valida que el estado de la aplicación sea consistente */ function validateApplicationState() { const errors = []; // Validar selecciones básicas if (!globalData.selectedYear) { errors.push('No hay año académico seleccionado'); } if (!globalData.selectedWorker) { errors.push('No hay trabajador seleccionado'); } // Validar datos cargados if (globalData.userAssignments.length === 0 && globalData.selectedYear && globalData.selectedWorker) { errors.push('No hay asignaciones presupuestales disponibles'); } // Validar consistencia de checkboxes const checkboxes = document.querySelectorAll('input[data-assignment-id]'); const assignmentsWithoutCheckbox = globalData.userAssignments.filter(assignment => { return !document.querySelector(`input[data-assignment-id="${assignment.assignment_id}"]`); }); if (assignmentsWithoutCheckbox.length > 0) { errors.push(`${assignmentsWithoutCheckbox.length} asignaciones sin checkbox correspondiente`); } return errors; } /** * Valida los datos antes de guardar */ function validateBeforeSave() { const baseErrors = validateApplicationState(); if (baseErrors.length > 0) { throw new Error('Estado de aplicación inválido: ' + baseErrors.join(', ')); } if (pendingChanges.size === 0) { throw new Error('No hay cambios para guardar'); } // Validar que el trabajador seleccionado existe const workerInfo = getWorkerInfo(globalData.selectedWorker); if (!workerInfo) { throw new Error('El trabajador seleccionado no está disponible'); } // Validar que todas las asignaciones en pendingChanges existen const invalidAssignments = []; pendingChanges.forEach((_, assignmentId) => { const assignment = globalData.userAssignments.find(a => a.assignment_id === assignmentId); if (!assignment) { invalidAssignments.push(assignmentId); } }); if (invalidAssignments.length > 0) { throw new Error(`Asignaciones inválidas en cambios pendientes: ${invalidAssignments.join(', ')}`); } return true; } // ===== UTILIDADES ===== /** * Obtiene un resumen del estado actual para debugging */ function getApplicationSummary() { return { selectedYear: globalData.selectedYear, selectedWorker: globalData.selectedWorker, totalAssignments: globalData.userAssignments.length, pendingChanges: pendingChanges.size, originalStateSize: originalState.size, validationErrors: validateApplicationState() }; } // Exponer funciones de debugging window.debugAssignRequesters = { ...window.debugAssignRequesters, getSummary: getApplicationSummary, validateState: validateApplicationState, getPendingChangesArray: () => Array.from(pendingChanges.entries()), getOriginalStateArray: () => Array.from(originalState.entries()) }; console.log('✅ Sistema de Designar Solicitantes - Funcionalidad de guardado completada');
